    const videosPerPage = 4;
    let currentPage = 1;
    let lastScrollY = window.scrollY;
    let currentFilteredVideos = videos;
    const navbar = document.getElementById('navbar');
    const aboutSection = document.getElementById('about');
    const requestPopup = document.getElementById('requestPopup');
    const requestForm = document.getElementById('requestForm');
    const requestProjectBtn = document.getElementById('requestProjectBtn');
    const closePopup = document.getElementById('closePopup');


    document.addEventListener('DOMContentLoaded', function() {
      const projectList = document.getElementById('project-list');
      if (!projectList) {
        console.error('Project list element not found!');
        return;
      }
      displayProjects(currentPage);
      setupEventListeners();
      setupSmoothScrolling();
      setupNavbarBehavior();
      setupPopupForm();
    });


    function setupEventListeners() {
      const searchInput = document.getElementById('searchInput');
      const clearSearch = document.getElementById('clearSearch');
      const searchSuggestions = document.getElementById('searchSuggestions');
      const suggestionList = document.getElementById('suggestionList');


      if (!searchInput || !clearSearch || !searchSuggestions || !suggestionList) {
        console.error('Search elements not found!');
        return;
      }


      searchInput.addEventListener('input', function() {
        clearSearch.style.display = this.value ? 'block' : 'none';
        aboutSection.classList.toggle('hidden', this.value.length > 0);
        searchProjects();
        updateSuggestions(this.value);
      });


      clearSearch.addEventListener('click', () => {
        searchInput.value = '';
        clearSearch.style.display = 'none';
        aboutSection.classList.remove('hidden');
        searchSuggestions.classList.remove('show');
        currentPage = 1;
        currentFilteredVideos = videos;
        displayProjects(currentPage);
      });


      suggestionList.addEventListener('click', (e) => {
        if (e.target.tagName === 'LI') {
          searchInput.value = e.target.textContent;
          searchSuggestions.classList.remove('show');
          searchProjects();
        }
      });


      document.addEventListener('click', (e) => {
        if (!searchInput.contains(e.target) && !searchSuggestions.contains(e.target)) {
          searchSuggestions.classList.remove('show');
        }
      });
    }


    function setupPopupForm() {
      if (!requestPopup || !requestForm || !requestProjectBtn || !closePopup) {

    function setupNavbarBehavior() {
      let lastScrollY = window.scrollY;
      window.addEventListener('scroll', () => {
        const currentScrollY = window.scrollY;
        if (currentScrollY > lastScrollY && currentScrollY > 100) {
          navbar.classList.add('hidden');
          navbar.classList.add('minimized');
        } else if (currentScrollY < lastScrollY || currentScrollY <= 100) {
          navbar.classList.remove('hidden');
          navbar.classList.toggle('minimized', currentScrollY > 100);
        }
        lastScrollY = currentScrollY;
      });
    }


    function setupSmoothScrolling() {
      document.querySelectorAll('a[href^="#"]').forEach(anchor => {
        anchor.addEventListener('click', function(e) {
          e.preventDefault();
          const targetId = this.getAttribute('href');
          if (targetId === '#') return;
          const targetElement = document.querySelector(targetId);
          if (targetElement) {
            window.scrollTo({
              top: targetElement.offsetTop - 100,
              behavior: 'smooth'
            });
          }
        });
      });
    }


    function updateSuggestions(searchTerm) {
      const suggestionList = document.getElementById('suggestionList');
      const searchSuggestions = document.getElementById('searchSuggestions');
      suggestionList.innerHTML = '';
      
      if (!searchTerm) {
        searchSuggestions.classList.remove('show');
        return;
      }


      const matchingProjects = videos.filter(video => 
        video.title.toLowerCase().includes(searchTerm.toLowerCase())
      );


      if (matchingProjects.length > 0) {
        matchingProjects.forEach(video => {
          const li = document.createElement('li');
          li.textContent = video.title;
          suggestionList.appendChild(li);
        });
        searchSuggestions.classList.add('show');
      } else {
        searchSuggestions.classList.remove('show');
      }
    }


    function searchProjects() {
      const searchTerm = document.getElementById('searchInput').value.toLowerCase();
      currentFilteredVideos = videos.filter(video => 
        video.title.toLowerCase().includes(searchTerm) || 
        video.description.toLowerCase().includes(searchTerm)
      );
      currentPage = 1;
      displayProjects(currentPage);
    }


    function displayProjects(page) {
      const projectList = document.getElementById('project-list');
      if (!projectList) {
        console.error('Project list element not found!');
        return;
      }


      const start = (page - 1) * videosPerPage;
      const end = start + videosPerPage;
      const paginatedVideos = currentFilteredVideos.slice(start, end);
      const searchTerm = document.getElementById('searchInput').value.toLowerCase();


      projectList.innerHTML = '';
      
      if (paginatedVideos.length === 0) {
        projectList.innerHTML = '<p class="no-projects" style="text-align: center; grid-column: 1/-1; padding: 2rem; color: #546e7a;">No projects found matching your search. Try different keywords.</p>';
        document.getElementById('pagination').innerHTML = '';
        return;
      }
      
      paginatedVideos.forEach(video => {
        const [brief, full] = video.description.split('||');
        const projectDiv = document.createElement('div');
        projectDiv.className = 'project';
        const isHighlighted = searchTerm && video.title.toLowerCase().includes(searchTerm);
        projectDiv.innerHTML = `
          <img src="${video.thumbnail}" alt="${video.title}" onerror="this.classList.add('error'); this.textContent='Thumbnail Not Available';">
          <div class="project-content">
            <h3 class="${isHighlighted ? 'highlight' : ''}">${video.title}</h3>
            <div class="description">
              <div class="brief">${brief}</div>
              ${full ? `<div class="full">${full}</div>` : ''}
              ${full ? '<div class="read-more" onclick="toggleDescription(this)">Read More</div>' : ''}
            </div>
            <div class="button-group">
              <a href="${video.url}" target="_blank" class="watch-btn"><i class="fas fa-play-circle"></i> Watch Tutorial</a>
              <a href="${video.download}" download class="download-btn"><i class="fas fa-download"></i> Download Code</a>
            </div>
          </div>
        `;
        projectList.appendChild(projectDiv);
      });


      updatePagination();
    }


    function toggleDescription(link) {
      const description = link.parentElement;
      const project = description.closest('.project');
      const isExpanded = description.classList.contains('show-full');


      description.classList.toggle('show-full');
      link.textContent = description.classList.contains('show-full') ? 'Read Less' : 'Read More';


      if (!isExpanded) {
        const contentHeight = description.querySelector('.full').offsetHeight;
        const projectContent = project.querySelector('.project-content');
        const buttonGroupHeight = projectContent.querySelector('.button-group').offsetHeight;
        const titleHeight = projectContent.querySelector('h3').offsetHeight;
        const padding = 32;
        const newHeight = 200 + titleHeight + contentHeight + buttonGroupHeight + padding;
        project.style.height = `${newHeight}px`;
      } else {
        project.style.height = 'auto';
      }
    }


    function updatePagination() {
      const pagination = document.getElementById('pagination');
      if (!pagination) {
        console.error('Pagination element not found!');
        return;
      }


      const totalPages = Math.ceil(currentFilteredVideos.length / videosPerPage);
      pagination.innerHTML = '';


      if (totalPages <= 1) return;


      if (currentPage > 1) {
        const prev = document.createElement('a');
        prev.href = '#';
        prev.innerHTML = '<i class="fas fa-chevron-left"></i> Previous';
        prev.addEventListener('click', (e) => {
          e.preventDefault();
          currentPage--;
          displayProjects(currentPage);
          window.scrollTo({ top: document.getElementById('projects').offsetTop - 100, behavior: 'smooth' });
        });
        pagination.appendChild(prev);
      }


      for (let i = 1; i <= totalPages; i++) {
        const pageLink = document.createElement('a');
        pageLink.href = '#';
        pageLink.textContent = i;
        pageLink.className = i === currentPage ? 'active' : '';
        pageLink.addEventListener('click', (e) => {
          e.preventDefault();
          currentPage = i;
          displayProjects(currentPage);
          window.scrollTo({ top: document.getElementById('projects').offsetTop - 100, behavior: 'smooth' });
        });
        pagination.appendChild(pageLink);
      }


      if (currentPage < totalPages) {
        const next = document.createElement('a');
        next.href = '#';
        next.innerHTML = 'Next <i class="fas fa-chevron-right"></i>';
        next.addEventListener('click', (e) => {
          e.preventDefault();
          currentPage++;
          displayProjects(currentPage);
          window.scrollTo({ top: document.getElementById('projects').offsetTop - 100, behavior: 'smooth' });
        });
        pagination.appendChild(next);
      }
    }
