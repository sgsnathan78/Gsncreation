const searchTerm = document.getElementById('searchInput').value.toLowerCase();
      currentFilteredVideos = videos.filter(video => 
        video.title.toLowerCase().includes(searchTerm) || 
        video.description.toLowerCase().includes(searchTerm)
      );
      currentPage = 1;
      displayProjects(currentPage);
    }


    function displayProjects(page) {
      const projectList = document.getElementById('project-list');
      if (!projectList) {
        console.error('Project list element not found!');
        return;
      }


      const start = (page - 1) * videosPerPage;
      const end = start + videosPerPage;
      const paginatedVideos = currentFilteredVideos.slice(start, end);
      const searchTerm = document.getElementById('searchInput').value.toLowerCase();


      projectList.innerHTML = '';
      
      if (paginatedVideos.length === 0) {
        projectList.innerHTML = '<p class="no-projects" style="text-align: center;color: #546e7a;">No projects found matching your search. Try different keywords.</p>';
        document.getElementById('pagination').innerHTML = '';
        return;
      }
      
      paginatedVideos.forEach(video => {
        const [brief, full] = video.description.split('||');
        const projectDiv = document.createElement('div');
        projectDiv.className = 'project';
        const isHighlighted = searchTerm && video.title.toLowerCase().includes(searchTerm);
        projectDiv.innerHTML = `
          <img src="${video.thumbnail}" alt="${video.title}" onerror="this.classList.add('error'); this.textContent='Thumbnail Not Available';">
          <div class="project-content">
            <h3 class="${isHighlighted ? 'highlight' : ''}">${video.title}</h3>
            <div class="description">
              <div class="brief">${brief}</div>
              ${full ? `<div class="full">${full}</div>` : ''}
              ${full ? '<div class="read-more" onclick="toggleDescription(this)">Read More</div>' : ''}
            </div>
            <div class="button-group">
              <a href="${video.url}" target="_blank" class="watch-btn"><i class="fas fa-play-circle"></i> Watch Tutorial</a>
              <a href="${video.download}" download class="download-btn"><i class="fas fa-download"></i> Download Code</a>
            </div>
          </div>
        `;
        projectList.appendChild(projectDiv);
      });


      updatePagination();
    }


    function toggleDescription(link) {
      const description = link.parentElement;
      const project = description.closest('.project');
      const isExpanded = description.classList.contains('show-full');


      description.classList.toggle('show-full');
      link.textContent = description.classList.contains('show-full') ? 'Read Less' : 'Read More';


      if (!isExpanded) {
        const contentHeight = description.querySelector('.full').offsetHeight;
        const projectContent = project.querySelector('.project-content');
        const buttonGroupHeight = projectContent.querySelector('.button-group').offsetHeight;
        const titleHeight = projectContent.querySelector('h3').offsetHeight;
        const padding = 32;
        const newHeight = 200 + titleHeight + contentHeight + buttonGroupHeight + padding;
        project.style.height = `${newHeight}px`;
      } else {
        project.style.height = 'auto';
      }
    }


    function updatePagination() {
      const pagination = document.getElementById('pagination');
      if (!pagination) {
        console.error('Pagination element not found!');
        return;
      }


      const totalPages = Math.ceil(currentFilteredVideos.length / videosPerPage);
      pagination.innerHTML = '';


      if (totalPages <= 1) return;


      if (currentPage > 1) {
        const prev = document.createElement('a');
        prev.href = '#';
        prev.innerHTML = '<i class="fas fa-chevron-left"></i> Previous';
        prev.addEventListener('click', (e) => {
          e.preventDefault();
          currentPage--;
          displayProjects(currentPage);
          window.scrollTo({ top: document.getElementById('projects').offsetTop - 100, behavior: 'smooth' });
        });
        pagination.appendChild(prev);
      }


      for (let i = 1; i <= totalPages; i++) {
        const pageLink = document.createElement('a');
        pageLink.href = '#';
        pageLink.textContent = i;
        pageLink.className = i === currentPage ? 'active' : '';
        pageLink.addEventListener('click', (e) => {
          e.preventDefault();
          currentPage = i;
          displayProjects(currentPage);
          window.scrollTo({ top: document.getElementById('projects').offsetTop - 100, behavior: 'smooth' });
        });
        pagination.appendChild(pageLink);
      }


      if (currentPage < totalPages) {
        const next = document.createElement('a');
        next.href = '#';
        next.innerHTML = 'Next <i class="fas fa-chevron-right"></i>';
        next.addEventListener('click', (e) => {
          e.preventDefault();
          currentPage++;
          displayProjects(currentPage);
          window.scrollTo({ top: document.getElementById('projects').offsetTop - 100, behavior: 'smooth' });
        });
        pagination.appendChild(next);
      }
    }
  </script>
